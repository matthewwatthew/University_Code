import java.util.ArrayList;
import java.util.Scanner;

public class Duplicates {

	public static void main(String[] args) {
		ArrayList<String> inputList = new ArrayList<String>();
		
		Scanner key = new Scanner(System.in);
		System.out.println("How many Strings would you like to enter?");
		int num = key.nextInt();
		key.nextLine();
		
		String str;
		for(int i = 0; i < num; i++) {
			System.out.println("What String would you like to add?");
			str = key.nextLine();
			inputList.add(str);
		}	
		
		System.out.println(inputList);
		System.out.println(removeDups(inputList));
		
		
		
		}
		
		public static ArrayList<String> removeDups(ArrayList<String> inputList) {
			
			ArrayList<String> dupsList = new ArrayList<String> (inputList.size());
			for  (int j = 0; j < inputList.size(); j++) {
				if (dupsList.contains(inputList.get(j))) {
					continue;
				}
				else { 
					dupsList.add(inputList.get(j));
				}
		}
			return dupsList;
	}

}
